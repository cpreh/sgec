function(
	add_sgec_library
	VARIANT
)
	if(
		"${VARIANT}"
		STREQUAL
		"STATIC"
	)
		set(
			TARGET_NAME
			sgec_static
		)
	else()
		set(
			TARGET_NAME
			sgec
		)
	endif()

	add_library(
		${TARGET_NAME}
		${VARIANT}
		${SGEC_FILES_ABS}
	)

	target_link_libraries(
		${TARGET_NAME}
		PRIVATE
		${fcppt_core_TARGET}
		${sgecore_TARGET}
		${sgeimage_TARGET}
		${sgeimagecolor_TARGET}
		${sgeinput_TARGET}
		${sgerenderer_TARGET}
		${sgesystems_TARGET}
		${sgeviewport_TARGET}
		${sgewindow_TARGET}
	)
endfunction()

		SHARED
	)
endif()

if(
	ENABLE_STATIC
)
	add_sgec_library(
		STATIC
	)
endif()
